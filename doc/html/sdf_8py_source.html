593  def _repetition_vector(self):
594 
595  # create a constraint problem solver object from ortools
596  solver = pywrapcp.Solver('Get repetition vector')
597 
598  topology_matrix = self.topology_matrix
599 
600  # Repetition vector P
601  # Each element is the number of invocatios of each SDF actor
602  # in one system iteration.
603  # The value of each repetition can be from value `1`
604  # to value `sys.maxsize` (the maximum integer value of the system).
605  # e.g in 32-bit operating system, `sys.maxsize` = \f$2^{31}-1\f$
606  P = [solver.IntVar(1, sys.maxsize, f'repetition count {actor.index}')
607  for actor in self.actors]
608 
609  # Add constraints
610  # sum of P element-wise multiply with
611  # one row (corresponding for one edge) in topology matrix
612  # should be zero.
613  # `ScalProd()` is a method in ortools,
614  # e.g A = [a0, a1, a2], B = [b0, b1, b2]
615  # ScalProd(A, B) = sum(a0*b0 + a1*b1 + a2*b2)
616  for row in topology_matrix:
617  each_row = solver.ScalProd(P, row)
618  # add a constraint that `each_row` should be 0
619  # to balance all the edges.
620  solver.Add(each_row == 0)
621 
622  # Create a database for solver
623  db = solver.Phase(P, # variables to search
624  solver.CHOOSE_FIRST_UNBOUND, # Branching strategy
625  solver.ASSIGN_MIN_VALUE) # Assignment strategy
626 
627  # Initialize a new search using the created database
628  solver.NewSearch(db)
629 
630  # Perform one search and assign a value to the search variables
631  # For search variable `var`,
632  # use `var.Value()` to get its current assigned value
633  solver.NextSolution()
634 
635  result = [int(var.Value()) for var in P]
636  # .Value() is for getting the current assigned value
637  # of the search variable.
638  # This method is from ortools
639 
640  return result
