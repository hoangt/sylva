141 def one_search(dse, log):
142  '''
143  Perform one search.
144 
145  Returns
146 
147  1. if a solution is found (True or False)
148  2. the solution
149  3. the search log
150  '''
151 
152  result = {}
153  performance = {}
154 
155  start_time = time.time()
156 
157  if dse.solver.NextSolution():
158 
159  end_time = time.time()
160 
161  log += 'Found one solution after %s branches\n' \
162  % dse.solver.Branches()
163  log += ' search time : %f\n' % ((end_time - start_time) / 1000)
164  log += ' FIMPs : %s\n' % len(dse.fimps)
165  log += ' area : %s\n' % dse.area.Value()
166  log += ' latency : %s\n' % dse.latency.Value()

173  result['input_end'] = [int(t.Value()) for t in dse.input_end]
174  result['output_start'] = [int(t.Value()) for t in dse.output_start]
175  result['buffer_end'] = [int(t.Value()) for t in dse.buffer_end]
176  result['extra_buffer'] = [int(t.Value()) for t in dse.extra_buffer]
177 
178  result['fimp_types'] = [-1 for f in dse.fimps]
179 
180  result['search time'] = (end_time - start_time) / 1000
181  result['fimps'] = len(dse.fimps)
182  result['area'] = dse.area.Value()
183  result['latency'] = dse.latency.Value()
184  result['energy'] = dse.energy.Value()
185  result['sample interval'] = dse.sample_interval.Value()
186 
187  for f in dse.fimps:
188  fimp_type = 0
189  for var in dse.fimp_selection_variable[f.index]:
190  if int(var.Value()) == 1:
191  result['fimp_types'][f.index] = fimp_type
192  fimp_type = fimp_type + 1
193 
194  return True, result # , log
195 
196  else:
197 
198  end_time = time.time()
199  log += 'No solution found after %s branches\n' \
200  % dse.solver.Branches()
201  log += ' search time : %f\n' % ((end_time - start_time) / 1000)
202  log.update()
203 
204  return False, result # , log
