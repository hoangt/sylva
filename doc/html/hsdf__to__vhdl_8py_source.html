100  # if the source port is wider: N to 1
101  def n_to_one(src_port, dest_port, n, src_signal_name, dest_signal_name):
102 
103  # destination is also std_logic_vector
104  if dest_port.type.name.lower().startswith('std_logic_vector'):
105  return [{'dest': dest_signal_name,

107  '( {i} * {width} - 1 downto 0)'.format(i=i, width=dest_port.type.size)}

115  # if the destination port is wider: 1 to N
116  def one_to_n(src_port, dest_port, n, src_signal_name, dest_signal_name):
117

119  if src_port.type.name.lower().startswith('std_logic_vector'):

122  'src': src_signal_name}
123  for i in range(n)]
124  # source is std_logic
125  else:
126  return [{'dest': dest_signal_name + '({i})'.format(i=i),
