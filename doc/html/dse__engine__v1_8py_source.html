288  def add_resource_dependency_constraints(self) :
289 
290  # computation and output buffer
291  for one_fimp in self.fimps :
292  if len(one_fimp.actors) > 1 :
293  for i, a in enumerate(one_fimp.actors[:-1]) :
294 
295  current = one_fimp.actors[i + 1].index
296  last = a.index
297 
298  # output start time > buffer end time
299  # or extra buffer is presented
300  normal = ( self.output_start[current] > self.buffer_end[last] ).Var()
301  extra_buffer = self.extra_buffer[last]
302  self.solver.Add(normal + extra_buffer > 0)
303 
304  # start time > end time for actors in a fimp
305  self.solver.Add( self.start[current] > self.end[last] )

307  def add_data_dependency_constraints(self) :
308 
309  for one_actor in self.actors :
310  if len(one_actor.next) > 0 :
311  input_end_list = [ self.end[one_actor.index] ]
312  for next_edge in one_actor.next :
313  # start time > end time
314  self.solver.Add( self.start[next_edge.dest_actor.index] \
315  > self.end[one_actor.index] )
316 
317  input_end_list.append(self.input_end[next_edge.dest_actor.index])
318 
319  # buffer end time = max(input end times, end time)
320  max_input_end = self.solver.Max(input_end_list).Var()
321  self.solver.Add( self.buffer_end[one_actor.index] == max_input_end )

351 def most_parallel_schedule(p, i, c, a) :
352 
353  '''
354  Find the most parallel schedule
355  when the amount of used FIMPs (c)
356  for an SDF actor with index (i) is given
357 
358  ---
359 
360  p: repetition vector
361  i: SDF actor index
362  c: FIMP count for SDF actor i
363  a: approximate numbers for p
364  '''
365 
366  ratio = int(p[i] / c)
367  # ratio is always integer
368  # since c is always one element of p[i]
369 
370  return [ ceil_in_list(a[index], int(p[index]/ratio))
371  for index in xrange(len(p)) ]

373 def ceil_in_list(ordered_number_list, number) :
374 
375  '''
376  Ceiling a given number in a given ordered number list.
377  '''
378 
379  for i, v in enumerate(ordered_number_list) :
380  if number <= v :
381  return v
382 
383  return ordered_number_list[-1]

399 def some_schedules( (sdf_actors, sdf_edges), size = None) :
400 
401  '''
402  Find a subset of all load balanced schedules for a given SDF graph.
403  '''
404 
405  if not size :
406  return all_schedules( (sdf_actors, sdf_edges) )
407 
408  p = get_repetition_count( (sdf_actors, sdf_edges) )
409  a = [ approximate_numbers(n) for n in p ]
410 
411  p_sorted = sorted([ (k, v) for k, v in enumerate(p)], key = lambda x : x[1], reverse = True)
412 
413  result = []
414  if len(a[p_sorted[0][0]]) < size :
415  for (k, v) in p_sorted :
416  result = schedules(p, a, k, result)
417  if len(result) >= size :
418  return result [:size]
419  return result
420  return evenly_pick(schedules(p, a, p[0][0], result), size)

422 def schedules(p, a, i, result = []) :
423  for c in a[i] :
424  schedule = most_parallel_schedule(p, i, c, a)

429 def assign_fimps(hsdf_actors, fimp_count_list) :
430 
431  '''
432  Create a relative schedule based on a given FIMP count list.
433  '''
434 
435  actors = {}
436  for a in hsdf_actors :
437  if a.base_actor.index not in actors.keys() :
438  actors[a.base_actor.index] = []
439  actors[a.base_actor.index].append(a)
440 
441  fimps = [ fimp(index = i)
442  for i in xrange(sum(fimp_count_list)) ]
443 
444  c = 0
445  # current FIMP index
446 
447  for i in actors.keys() : # for each SDF actor
448 
449  f = 0
450  # current FIMP index for one SDF actor
451 
452  for a in actors[i] :
453  # for each HSDF actor
454  if f == fimp_count_list[i] :
455  c -= fimp_count_list[i]
456  f = 0
457 
458  fimps[c].name = a.name
459  fimps[c].add_actor(a)
460  c += 1
461  f += 1
462 
463  return fimps
