42

44  minvalue = 0

46 
47  else :
48  raise NotImplementedError

50  if value > maxvalue :
51  raise ValueError( 'Current value (%s) is larger than the maximum value (%s)' \
52  % (value, maxvalue) )
53  if value < minvalue :
54  raise ValueError( 'Current value (%s) is smaller than the minimum value (%s)' \
55  % (value, minvalue) )
56
